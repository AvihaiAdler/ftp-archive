set(LOGGER_UNIT_TESTS logger_sanity)

foreach(TEST ${LOGGER_UNIT_TESTS})
  
add_executable(${TEST} ${TEST}.c)
add_test(NAME ${TEST} COMMAND ${PROJECT_SOURCE_DIR}/build/tests/logger/${TEST})

target_compile_options(${TEST} PRIVATE -Wall -Wextra -pedantic -O3)
target_link_options(${TEST} PRIVATE -fsanitize=address,undefined -lpthread)

target_include_directories(${TEST} PRIVATE ${CMAKE_SOURCE_DIR})

# import the libraries
# logger
add_library(liblogger STATIC IMPORTED)
set_target_properties(
  liblogger
  PROPERTIES
  IMPORTED_LOCATION ${CMAKE_SOURCE_DIR}/lib/liblogger.a
  INTERFACE_INCLUDE_DIRECTORIES ${CMAKE_SOURCE_DIR}/lib/logger/include
)

target_link_libraries(
  ${TEST} PRIVATE 
  liblogger
)


endforeach(TEST)
