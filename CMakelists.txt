cmake_minimum_required(VERSION 3.10)
project(ftp VERSION 0.0.1)

set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED true)

include(CTest)
add_subdirectory(lib/generics)
add_subdirectory(lib/properties)
add_subdirectory(lib/logger)
add_subdirectory(lib/thread_pool)
add_subdirectory(lib/payload)
add_subdirectory(server)
add_subdirectory(client)

# copy the final executable into /bin
file(COPY ${CMAKE_BINARY_DIR}/server/ftpd DESTINATION ${CMAKE_SOURCE_DIR}/bin)
